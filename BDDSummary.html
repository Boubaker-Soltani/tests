🔥 التطبيع Normalisation

1NF

2NF

3NF

لا يوجد اعتماد انتقالي

كل اعتماد وظيفي مصدره مفتاح

💻 أهم أوامر SQL

✅ SELECT

SELECT * FROM EMP;

✅ WHERE

SELECT * FROM EMP WHERE SALARY > 1000;

✅ DISTINCT

SELECT DISTINCT JOB FROM EMP;

✅ ORDER BY

SELECT * FROM EMP ORDER BY SALARY DESC;

✅ LIKE

النمط المعنى
'A%' يبدأ بـ A
'%A' ينتهي بـ A
'%A%' يحتوي A
'_A' حرف ثم A

✅ BETWEEN

SELECT * FROM EMP WHERE SALARY BETWEEN 1000 AND 3000;

✅ IN

SELECT * FROM EMP WHERE DEPT IN (10,20);

✅ EXISTS

SELECT * FROM ETUDIANT E WHERE EXISTS ( SELECT * FROM INSCRIPTION I WHERE E.ID = I.ID );

✅ NOT EXISTS

SELECT * FROM ETUDIANT E WHERE NOT EXISTS ( SELECT * FROM INSCRIPTION I WHERE E.ID = I.ID );

📊 الدوال التجميعية

الدالة المعنى
COUNT() عدد الصفوف
AVG() المتوسط
SUM() المجموع
MIN() أصغر قيمة
MAX() أكبر قيمة

✅ GROUP BY

SELECT Dept, AVG(Salary) FROM EMP GROUP BY Dept;

✅ HAVING

SELECT Dept, AVG(Salary) FROM EMP GROUP BY Dept HAVING AVG(Salary) > 2000;
📌 WHERE تطبق قبل GROUP BY
📌 HAVING تطبق بعد GROUP BY
